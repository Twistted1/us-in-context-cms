const toggleProjectsTask = useCallback((taskId) =
  setCompletedProjectsTasks(prev =
    const newCompleted = [...prev]; 
    const index = newCompleted.indexOf(taskId); 
    if (index > -1) { 
      newCompleted.splice(index, 1); 
    } else { 
      newCompleted.push(taskId); 
    } 
    return newCompleted; 
  }); 
}, []);
